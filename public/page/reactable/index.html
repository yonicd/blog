GitHub Activity Reactable

reactable::reactable(tbl_md, 
          pagination = FALSE, 
          highlight = TRUE, 
          height = 500,
          defaultSortOrder = "desc",
          defaultSorted = c("views"),
          rowStyle = function(index) {
            if (tbl_md[index, "type"] =='owner') {
              list(background = "rgba(0, 0, 0, 0.05)")
            }
          }
)
